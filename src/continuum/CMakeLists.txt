        ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/*.cc)

    add_library(continuum STATIC ${CONTINUUM_SRC})

        INTERFACE numeric femsolver_options)

        INTERFACE ${CMAKE_CURRENT_SOURCE_DIR})
